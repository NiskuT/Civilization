500 #define _VKI_IOC(dir,type,nr,size) \
501  (((dir) << _VKI_IOC_DIRSHIFT) | \
502  ((type) << _VKI_IOC_TYPESHIFT) | \
503  ((nr) << _VKI_IOC_NRSHIFT) | \
504  ((size) << _VKI_IOC_SIZESHIFT))

506 /* used to create numbers */
507 #define _VKI_IO(type,nr) _VKI_IOC(_VKI_IOC_NONE,(type),(nr),0)
508 #define _VKI_IOR(type,nr,size) _VKI_IOC(_VKI_IOC_READ,(type),(nr),(_VKI_IOC_TYPECHECK(size)))
