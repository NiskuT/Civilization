73 /* Controls how much history to collect, in order to show conflicting
74  accesses. There are three levels:
75 
76  0: "none": don't collect any history. Fastest, but means we can
77  only show one of the two stacks in a race.
78 
79  1: "approx": collect one stack trace per (notional) segment, that
80  is, collect a stack trace for a thread every time its vector
81  clock changes. This facilitates showing the bounds of the
82  conflicting segment(s), with relatively small overhead.
83 
84  2: "full": collect a stack trace every time the constraints for a
85  location change. This facilitates showing both stack traces in
86  a race, although can be very expensive in time and space
87  (depends on the rate that threads "drag" locations across
88  vector-clock-change boundaries ("dragovers"). This involves
89  collecting and storing large numbers of call stacks just in case
90  we might need to show them later, and so is expensive (although
91  very useful). */
92 extern UWord HG_(clo_history_level);
93 
94 /* For full history level, determines how the stack trace is computed.
95  no : a stacktrace is always computed from scratch, typically
96  using the unwind information.
97  yes : a stacktrace might be derived from a previous captured stacktrace,
98  by applying a delta to the inner frame. This can significantly
99  speed up --history-level=full. Typically, a delta can be applied
100  if there was no 'flow call control' instruction (call, return, ...)
101  executed since the previous captured stacktrace. */
102 extern Bool HG_(clo_delta_stacktrace);
103 
104 /* When doing "full" history collection, this determines the size of
105  the conflicting-access cache, measured in terms of maximum possible
106  number of elements in the previous-access map. Must be between 10k
107  amd 10 million. Default is 1 million. */
108 extern UWord HG_(clo_conflict_cache_size);
109 
110 /* Sanity check level. This is an or-ing of
111  SCE_{THREADS,LOCKS,BIGRANGE,ACCESS,LAOG}. */
112 extern UWord HG_(clo_sanity_flags);
113 
114 /* Treat heap frees as if the memory was written immediately prior to
115  the free. This shakes out races in which memory is referenced by
116  one thread, and freed by another, and there's no observable
117  synchronisation event to guarantee that the reference happens
118  before the free. */
119 extern Bool HG_(clo_free_is_write);
120 
121 /* Controls the application of VTS pruning. There are three levels:
122 
123  0: "never": VTS pruning is never done
124 
125  1: "auto": (the default): VTS pruning is sometimes done after VTS
126  GCs, just frequently enough to keep space use under control, as
127  determined by heuristics in libhb_core.c.
128 
129  2: "always": VTS pruning is done after every VTS GC. This is
130  mostly a big time waster, but minimises space use. */
131 extern UWord HG_(clo_vts_pruning);
132 
133 /* When False, race checking ignores memory references which are to
134  the stack, which speeds things up a bit. Default: True. */
135 extern Bool HG_(clo_check_stack_refs);
