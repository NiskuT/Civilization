61 // This is both the minimum payload size of a malloc'd block, and its
62 // minimum alignment. Must be a power of 2 greater than 4, and should be
63 // greater than 8.
64 #if defined(VGP_x86_linux) || \
65  defined(VGP_arm_linux) || \
66  defined(VGP_mips32_linux) || \
67  (defined(VGP_mips64_linux) && defined(VGABI_N32)) || \
68  defined(VGP_nanomips_linux) || \
69  defined(VGP_x86_solaris)
70 # define VG_MIN_MALLOC_SZB 8
71 // Nb: We always use 16 bytes for Darwin, even on 32-bits, so it can be used
72 // for any AltiVec- or SSE-related type. This matches the Darwin libc.
73 // Also, use 16 bytes for any PPC variant, since 16 is required to make
74 // Altiveccery work right.
75 #elif defined(VGP_amd64_linux) || \
76  defined(VGP_ppc32_linux) || \
77  defined(VGP_ppc64be_linux) || \
78  defined(VGP_ppc64le_linux) || \
79  defined(VGP_s390x_linux) || \
80  (defined(VGP_mips64_linux) && !defined(VGABI_N32)) || \
81  defined(VGP_x86_freebsd) || \
82  defined(VGP_amd64_freebsd) || \
83  defined(VGP_x86_darwin) || \
84  defined(VGP_amd64_darwin) || \
85  defined(VGP_arm64_linux) || \
86  defined(VGP_amd64_solaris)
87 # define VG_MIN_MALLOC_SZB 16
88 #else
