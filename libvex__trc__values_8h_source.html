38 /* Magic values that the guest state pointer might be set to when
39  returning to the dispatcher. The only other legitimate value is to
40  point to the start of the thread's VEX guest state.
41 
42  This file may get included in assembly code, so do not put
43  C-specific constructs in it.
44 
45  These values should be 61 or above so as not to conflict
46  with Valgrind's VG_TRC_ values, which are 60 or below.
47 */
48 
49 #define VEX_TRC_JMP_INVALICACHE 61 /* invalidate icache (translations)
50  before continuing */
51 #define VEX_TRC_JMP_FLUSHDCACHE 103 /* flush dcache before continuing */
52 
53 #define VEX_TRC_JMP_NOREDIR 81 /* jump to undirected guest addr */
54 #define VEX_TRC_JMP_SIGTRAP 85 /* deliver trap (SIGTRAP) before
55  continuing */
56 #define VEX_TRC_JMP_SIGSEGV 87 /* deliver segv (SIGSEGV) before
57  continuing */
58 #define VEX_TRC_JMP_SIGBUS 93 /* deliver SIGBUS before continuing */
59 #define VEX_TRC_JMP_SIGFPE 105 /* deliver SIGFPE before continuing */
60 
61 #define VEX_TRC_JMP_SIGFPE_INTDIV 97 /* deliver SIGFPE (integer divide
62  by zero) before continuing */
63 
64 #define VEX_TRC_JMP_SIGFPE_INTOVF 99 /* deliver SIGFPE (integer overflow)
65  before continuing */
66 
67 #define VEX_TRC_JMP_SIGILL 101 /* deliver SIGILL (Illegal instruction)
68  before continuing */
69 
70 #define VEX_TRC_JMP_EMWARN 63 /* deliver emulation warning before
71  continuing */
72 #define VEX_TRC_JMP_EMFAIL 83 /* emulation fatal error; abort system */
73 
74 #define VEX_TRC_JMP_CLIENTREQ 65 /* do a client req before continuing */
75 #define VEX_TRC_JMP_YIELD 67 /* yield to thread sched
76  before continuing */
77 #define VEX_TRC_JMP_NODECODE 69 /* next instruction is not decodable */
78 #define VEX_TRC_JMP_MAPFAIL 71 /* address translation failed */
79 
80 #define VEX_TRC_JMP_SYS_SYSCALL 73 /* do syscall before continuing */
81 #define VEX_TRC_JMP_SYS_INT32 75 /* do syscall before continuing */
82 #define VEX_TRC_JMP_SYS_INT128 77 /* do syscall before continuing */
83 #define VEX_TRC_JMP_SYS_INT129 89 /* do syscall before continuing */
84 #define VEX_TRC_JMP_SYS_INT130 91 /* do syscall before continuing */
85 #define VEX_TRC_JMP_SYS_INT145 111 /* do syscall before continuing */
86 #define VEX_TRC_JMP_SYS_INT210 113 /* do syscall before continuing */
87 
88 #define VEX_TRC_JMP_SYS_SYSENTER 79 /* do syscall before continuing */
89 
90 #define VEX_TRC_JMP_BORING 95 /* return to sched, but just
91  keep going; no special action */
