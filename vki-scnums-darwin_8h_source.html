36 // osfmk/mach/i386/syscall_sw.h
37 
38 // There are two syscall number encodings in Darwin.
39 //
40 // The 64-bit encoding is that the top 8-bits are the syscall class. The low
41 // 24 are the syscall number (index) within that class.
42 //
43 // The 32-bit encoding is that the syscall number (index) is stored as-is and
44 // the syscall class is encoded as the argument to the 'int' instruction used
45 // to trigger the syscall:
46 // - 0x80: Unix
47 // - 0x81: Mach
48 // - 0x82: Machine-dependent
49 // - 0x83: Diagnostic
50 // Furthermore, just to make life interesting, for Mach traps the number is
51 // negative.
52 //
53 // Within Valgrind we only use the 64-bit encoding -- on 32-bit systems, we
54 // convert any syscall number to 64-bit encoding when we receive it, and
55 // convert back with VG_DARWIN_SYSNO_FOR_KERNEL when passing any syscall
56 // number back to the kernel (__NR_something shouldn't be passed directly to
57 // the kernel).
58 //
59 // Hack: x86 `int $0x80` (unix, 64-bit result) are special.
60 // [I haven't worked out why... --njn]
61 
62 #define VG_DARWIN_SYSCALL_CLASS_SHIFT 24
63 #define VG_DARWIN_SYSCALL_CLASS_MASK (0xFF << VG_DARWIN_SYSCALL_CLASS_SHIFT)
64 #define VG_DARWIN_SYSCALL_NUMBER_MASK (~VG_DARWIN_SYSCALL_CLASS_MASK)
65 
66 #define VG_DARWIN_SYSCALL_CLASS_NONE 0 /* Invalid */
67 #define VG_DARWIN_SYSCALL_CLASS_MACH 1 /* Mach */
68 #define VG_DARWIN_SYSCALL_CLASS_UNIX 2 /* Unix/BSD */
69 #define VG_DARWIN_SYSCALL_CLASS_MDEP 3 /* Machine-dependent */
70 #define VG_DARWIN_SYSCALL_CLASS_DIAG 4 /* Diagnostics */
71 
72 // Macros for encoding syscall numbers in the 64-bit encoding scheme.
73 #define VG_DARWIN_SYSCALL_CONSTRUCT_MACH(syscall_number) \
74  ((VG_DARWIN_SYSCALL_CLASS_MACH << VG_DARWIN_SYSCALL_CLASS_SHIFT) | \
75  (VG_DARWIN_SYSCALL_NUMBER_MASK & (syscall_number)))
76 
77 #define VG_DARWIN_SYSCALL_CONSTRUCT_UNIX(syscall_number) \
78  ((VG_DARWIN_SYSCALL_CLASS_UNIX << VG_DARWIN_SYSCALL_CLASS_SHIFT) | \
79  (VG_DARWIN_SYSCALL_NUMBER_MASK & (syscall_number)))
80 
81 #define VG_DARWIN_SYSCALL_CONSTRUCT_MDEP(syscall_number) \
82  ((VG_DARWIN_SYSCALL_CLASS_MDEP << VG_DARWIN_SYSCALL_CLASS_SHIFT) | \
83  (VG_DARWIN_SYSCALL_NUMBER_MASK & (syscall_number)))
84 
85 #define VG_DARWIN_SYSCALL_CONSTRUCT_DIAG(syscall_number) \
86  ((VG_DARWIN_SYSCALL_CLASS_DIAG << VG_DARWIN_SYSCALL_CLASS_SHIFT) | \
87  (VG_DARWIN_SYSCALL_NUMBER_MASK & (syscall_number)))
88 
89 
90 /* Macros for decoding syscall numbers from the 64-bit encoding scheme. */
91 #define VG_DARWIN_SYSNO_INDEX(sysno) ((sysno) & VG_DARWIN_SYSCALL_NUMBER_MASK)
92 #define VG_DARWIN_SYSNO_CLASS(sysno) ((sysno) >> VG_DARWIN_SYSCALL_CLASS_SHIFT)
93 
94 
95 /* Macros for converting syscall numbers to the form expected by the kernel.*/
96 #if defined(VGA_x86)
97  // This converts the 64-bit syscall number encoding, which we use
98  // throughout Valgrind, into the 32-bit syscall number encoding, which is
99  // suitable for passing to the (32-bit) kernel.
100 # define VG_DARWIN_SYSNO_FOR_KERNEL(sysno) \
101  ((VG_DARWIN_SYSNO_CLASS(sysno) == VG_DARWIN_SYSCALL_CLASS_MACH) \
102  ? -VG_DARWIN_SYSNO_INDEX(sysno) \
103  : VG_DARWIN_SYSNO_INDEX(sysno) \
104  )
105 
106 #elif defined(VGA_amd64)
107  // For 64-bit systems, we don't need to do anything to the syscall number.
108 # define VG_DARWIN_SYSNO_FOR_KERNEL(sysno) (sysno)
109 
110 #else
111 # error Unknown architecture
112 #endif
