310 /* Macro definitions for opcodes that are not generally available.
311 
312  The values to be encoded in those fields must be integer values in
313  hexadecimal notation without a leading 0x.
314  E.g. VRX_VXBD(e7, 1, 0, 3, 0000, 0, 06) is equal to "vl %%v1, 0(%%r3)\n\t"
315 */
316 #define VRX_VXBD(op1, v1, x2, b2, d2, rxb, op2) \
317  ".short 0x" #op1 #v1 #x2 "\n\t .int 0x" #b2 #d2 "0" #rxb #op2 "\n\t"
318 #define VRR_VVVMM(op1, v1, v2, v3, m5, m4, rxb, op2) \
319  ".short 0x" #op1 #v1 #v2 "\n\t .int 0x" #v3 "0" #m5 "0" #m4 #rxb #op2 "\n\t"
320 
321 #define VL(v1, x2, b2, d2, rxb) VRX_VXBD(e7, v1, x2, b2, d2, rxb, 06)
322 #define VST(v1, x2, b2, d2, rxb) VRX_VXBD(e7, v1, x2, b2, d2, rxb, 0e)
323 #define VPKS(v1, v2, v3, m4, m5, rxb) VRR_VVVMM(e7, v1, v2, v3, m5, m4, rxb, 97)
