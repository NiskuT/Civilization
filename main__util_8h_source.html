133 /* Allocated memory as returned by LibVEX_Alloc will be aligned on this
134  boundary. */
135 #define REQ_ALIGN 8
136 
137 #if defined(ENABLE_INNER)
138 #define VEX_REDZONE_SIZEB (2*REQ_ALIGN)
139 #endif
140 
141 static inline void* LibVEX_Alloc_inline ( SizeT nbytes )
142 {
143  struct align {
144  char c;
145  union {
146  char c;
147  short s;
148  int i;
149  long l;
150  long long ll;
151  float f;
152  double d;
153  /* long double is currently not used and would increase alignment
154  unnecessarily. */
155  /* long double ld; */
156  void *pto;
157  void (*ptf)(void);
158  } x;
159  };
160 
161  /* Make sure the compiler does no surprise us */
162  vassert(offsetof(struct align,x) <= REQ_ALIGN);
163 
164 #if 0
165  /* Nasty debugging hack, do not use. */
166  return malloc(nbytes);
167 #else
168  HChar* curr;
169  HChar* next;
170  SizeT ALIGN;
171  ALIGN = offsetof(struct align,x) - 1;
172  curr = private_LibVEX_alloc_curr;
173  next = curr + ((nbytes + ALIGN) & ~ALIGN);
174  INNER_REQUEST(next += 2 * VEX_REDZONE_SIZEB);
175  if (next >= private_LibVEX_alloc_last)
176  private_LibVEX_alloc_OOM();
177  private_LibVEX_alloc_curr = next;
178  INNER_REQUEST(curr += VEX_REDZONE_SIZEB);
179  INNER_REQUEST(VALGRIND_MEMPOOL_ALLOC(private_LibVEX_alloc_first,
180  curr, nbytes));
181  return curr;
182 #endif
183 }
